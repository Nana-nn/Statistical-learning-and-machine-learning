def pca(X_std,Y_tr,test_X,test_y):

    W_srt = W.tolist()
    W_srt.sort(reverse=True)
    # ()
    summ = 0
    W_sum = [0]
    for i in range(p):
        summ += W_srt[i]
        W_sum.append(summ / p)
    #80%
    c_pc = 0.8
    cnt = True
    thres = p * c_pc
    while cnt:
        W_sum = 0
        W_summ = W_srt[0]

            W_sum += W_srt[i]
            W_summ += W_srt[i+1]
            if (W_sum < thres) & (W_summ >= thres):
                cnt = False
                break
            elif i == p - 2:
                cnt = False
                k1 = i + 1
                break
    k1 = k1 + 1
