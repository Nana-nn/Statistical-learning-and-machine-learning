#  x_0 E(y_0) 
def confidence_interval(x0):
    x0 = np.array(x0)
    Y0 = np.dot(x0.T, beta_hat)
    delta0 = tVal * sigma * np.sqrt(x0.T @ C @ x0)
    Y0_int = [Y0 - delta0, Y0 + delta0]
    return Y0_int

#  x_0 y_0 
def prediction_interval(x0):
    x0 = np.array(x0)
    Y0 = np.dot(x0.T, beta_hat)
    delta1 = tVal * sigma * np.sqrt(1 + x0.T @ C @ x0)
    Y0_int = [Y0 - delta1, Y0 + delta1]
    return Y0_int
